<script>

    async function submitJob() {


        const pickup = document.getElementById("pickup").value;

        const dropoff = document.getElementById("dropoff").value;

        const description = document.getElementById("description").value;

        const messageBox = document.getElementById("message");


        if (!pickup || !dropoff || !description) {

            messageBox.style.display = "block";

            messageBox.className = "message error";

            messageBox.innerText = "All fields are required.";

            return;

        }


        try {

            const response = await fetch("http://YOUR_SERVER_URL_HERE/jobs", {

                method: "POST",

                headers: { "Content-Type": "application/json" },

                body: JSON.stringify({

                    pickup_location: pickup,

                    dropoff_location: dropoff,

                    load_description: description

                })

            });


            const result = await response.json();


            if (response.ok) {

                messageBox.style.display = "block";

                messageBox.className = "message success";

                messageBox.innerText = "Job created successfully! ID: " + result.id;


                document.getElementById("pickup").value = "";

                document.getElementById("dropoff").value = "";

                document.getElementById("description").value = "";


            } else {

                throw new Error(result.detail || "Failed to create job");

            }


        } catch (err) {

            messageBox.style.display = "block";

            messageBox.className = "message error";

            messageBox.innerText = "Error: " + err.message;

        }

    }

</script>
